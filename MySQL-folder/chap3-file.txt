-- WHERE (CONDITION)
-- LEVEL 1
SELECT * FROM dim_customer;

SELECT
	* 
FROM
    dim_customer
WHERE 
	(GENDER = 'F') AND (COUNTRY = 'FRANCE') AND (join_date > '2022-01-01');
-- pruning means when u want to filter out your data
-- ONLY 1 WHERE CLAUSE


-- LEVEL 2 (AND/OR)
SELECT
	* 
FROM
    dim_customer
WHERE 
	(GENDER = 'F') AND ((COUNTRY = 'FRANCE') OR (join_date > '2022-01-01'));
-- 2 CONDITIONS GENDER AND COUNTRY+JOINDATE


-- LIKE

-- LEVEL 1
SELECT
	*
FROM
	dim_customer
WHERE
	first_name LIKE 'T%';
    
-- LEVEL 2
SELECT
	*
FROM
	dim_customer
WHERE
	first_name LIKE 'T__F%Y';



-- SORTING
SELECT * FROM dim_product;

-- DEFAULT ASC 
SELECT 
	* 
FROM 
	dim_product
ORDER BY
	unit_price;

-- DESC DESCENDING ORDER
SELECT 
	* 
FROM 
	dim_product
ORDER BY
	unit_price DESC;

-- ASC ASCENDING ORDER
SELECT 
	* 
FROM 
	dim_product
ORDER BY
	unit_price ASC;

-- SORTING 3 MOST EXPENSIVE
SELECT 
	* 
FROM 
	dim_product
ORDER BY
	unit_price DESC
LIMIT 3;
