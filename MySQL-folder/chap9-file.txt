-- SUBQUERIES

-- LEVEL 1
SELECT AVG(UNIT_PRICE) FROM DIM_PRODUCT;
-- 495.79

SELECT 
	*
FROM
	DIM_PRODUCT
WHERE 
	UNIT_PRICE > (SELECT AVG(UNIT_PRICE) FROM DIM_PRODUCT);
    
-- IT SHOWS UNIT PRICE > AVG UNIT PRICE


-- LEVEL 2
SELECT 
	*
FROM
(
	SELECT 
	*
FROM
	DIM_PRODUCT
WHERE 
	UNIT_PRICE > (SELECT AVG(UNIT_PRICE) FROM DIM_PRODUCT)
) AS SUBQUERY_TABLE
WHERE
	PRODUCT_NAME = 'FIGURE METHOD'
