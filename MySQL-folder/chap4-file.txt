-- JOINS
CREATE TABLE ORDERS
(
	O_ID INT,
    CUST_ID INT,
    PRICE INT
);

INSERT INTO ORDERS
VALUES
(1,101,1000),
(2,201,1100),
(3,501,1200);

CREATE TABLE CUSTOMERS
(
	ID INT,
    NAME VARCHAR(100),
    EMAIL VARCHAR(100)
);

INSERT INTO CUSTOMERS
VALUES
(101,'LOVE','AA'),
(201,'ANSH','BB'),
(301,'LAMBA','CC');


-- INNER JOIN
SELECT
	O.*,
    C.ID
FROM 
	ORDERS O
INNER JOIN
	CUSTOMERS C
	on
		O.CUST_ID = C.ID;

-- LEFT JOIN
-- FROM IS LEFT, LEFT JOIN IS RIGHT
SELECT
	*
FROM 
	ORDERS O
LEFT JOIN
	CUSTOMERS C
	on
		O.CUST_ID = C.ID;
        
-- RIGHT JOIN
SELECT
	*
FROM 
	ORDERS O
RIGHT JOIN
	CUSTOMERS C
	on
		O.CUST_ID = C.ID;
        
-- FULL JOIN (not supported in this mysql)
-- USE UNION INSTEAD, FULL JOIN = LEFT + RIGHT

SELECT
	*
FROM 
	ORDERS O
LEFT JOIN
	CUSTOMERS C
	on
		O.CUST_ID = C.ID
        
UNION

SELECT
	*
FROM 
	ORDERS O
RIGHT JOIN
	CUSTOMERS C
	on
		O.CUST_ID = C.ID;
